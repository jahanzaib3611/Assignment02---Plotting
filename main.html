plt.text(mean_val + 0.5, 20, f'Mean: {mean_val:.2f}', color='red', fontsize=12)

# for median
plt.axvline(median_val, color='green', linestyle='dashed', linewidth=2, label=f'Median: {median_val:.2f}')
plt.text(median_val + 0.5, 18, f'Median: {median_val:.2f}', color='green', fontsize=12)

# setting title, labels,legend and font size for better readability
